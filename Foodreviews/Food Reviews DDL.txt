CREATE TABLE Restaurants
(
RestaurantID VARCHAR(3) PRIMARY KEY,
RestaurantName VARCHAR(100) NOT NULL,
RestaurantPostcode VARCHAR(10) NOT NULL,
RestaurantRating REAL
);

CREATE TABLE Users
(
UserName VARCHAR(25) PRIMARY KEY,
UserEmail VARCHAR(100) UNIQUE NOT NULL,
UserPassword VARCHAR(100) NOT NULL,
UserFirstName VARCHAR(100) NOT NULL,
UserLastName VARCHAR(100) NOT NULL,
UserPhone VARCHAR(20)
);

CREATE TABLE Tags
(
Tag VARCHAR(25) PRIMARY KEY
);

CREATE TABLE RestaurantTags
(
Tag VARCHAR(25),
RestaurantID VARCHAR(3),
PRIMARY KEY (RestaurantID, Tag),
FOREIGN KEY (RestaurantID) REFERENCES Restaurants(RestaurantID),
FOREIGN KEY (Tag) REFERENCES Tags(Tag)
);

CREATE TABLE Reviews
(
ReviewNumber INT PRIMARY KEY IDENTITY(1,1),
UserName VARCHAR(25),
RestaurantID VARCHAR(3),
ReviewDate DATE,
ReviewStarRating INT,
ReviewText VARCHAR(5000),
FOREIGN KEY (RestaurantID) REFERENCES Restaurants(RestaurantID),
FOREIGN KEY (UserName) REFERENCES Users(UserName)
);
